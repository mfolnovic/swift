Class ModelBase

Description
Description | Vars (details) | Methods (details)

Base model, allows creating ActiveRecord like models

Example:

  1.  // in controller
  2.  
  3.  // creating new instance of model User
  4.  $user = $this -> model( 'user' );
  5.  
  6.  // finding a user by username
  7.  $user = $this -> model( 'user' ) -> find_by_username( 'username' );
  8.  
  9.  // creating new user from POST data
  10.  $user = $this -> model( 'user', $this -> data ) -> save();
  11.  
  12.  // updating record from POST data
  13.  $user = $this -> model( 'user' ) -> find_by_ID( $this -> data[ "id" ] ) -> values( $this -> data ) -> save();
  14.  
  15.  // deleting record
  16.  $this -> model( 'user' ) -> find_by_ID( $this -> data[ "id"]  ) -> delete();
  17.  
  18.  // using other relations
  19.  $user = $this -> model( 'user' ) -> where( "`username` like '%multi%'" ) -> order( 'ID desc' ) -> select( 'username, password' );
  20.  // This will create query (but not run, until you want to get any field/row): SELECT username, password FROM user WHERE `username` like '%multi%' ORDER BY `ID` desc;
  21.  
  22.  // you can also access attributes
  23.  $user = $this -> model( 'user' ) -> first();
  24.  echo $user -> username;
